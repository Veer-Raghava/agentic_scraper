# ── Injectable topic-specific patterns ───────────────────────────────────────
# Call update_patterns() once per run (from pipeline.py) to inject LLM-generated
# patterns.  These supplement — never replace — the built-in academic patterns.

_extra_section_headers: list[str] = []          # regex fragments for section headings
_extra_url_patterns: dict[str, re.Pattern] = {} # label → compiled regex for URLs/IDs


def update_patterns(
    section_headers: list[str] | None = None,
    url_patterns: dict[str, str] | None = None,
) -> None:
    """
    Inject topic-specific patterns generated by the LLM.
    section_headers – list of lowercase regex fragments (e.g. r"data\\s+sources")
    url_patterns    – dict of {label: raw_regex_string}
    Both are additive: built-in patterns still apply.
    """
    global _extra_section_headers, _extra_url_patterns
    if section_headers:
        _extra_section_headers = [str(h).lower() for h in section_headers]
    if url_patterns:
        compiled: dict[str, re.Pattern] = {}
        for label, pattern in url_patterns.items():
            try:
                compiled[str(label)] = re.compile(str(pattern), re.I)
            except re.error as exc:
                console.print(f"  [dim yellow]⚠ Invalid URL pattern '{pattern}': {exc}[/dim yellow]")
        _extra_url_patterns = compiled



DOI_PATTERN = re.compile(r"10\.\d{4,}/[^\s,;\"'<>]+")
REPO_PATTERNS = {
    "Zenodo":   re.compile(r"zenodo\.org/record[s]?/\d+", re.I),
    "Figshare": re.compile(r"figshare\.com/\S+", re.I),
    "GitHub":   re.compile(r"github\.com/[\w\-]+/[\w\-]+", re.I),
    "Dryad":    re.compile(r"datadryad\.org/\S+", re.I),
    "Kaggle":   re.compile(r"kaggle\.com/\S+", re.I),
    "OSF":      re.compile(r"osf\.io/\S+", re.I),
}
ACCESSION_PATTERNS = {
    "GEO":       re.compile(r"GSE\d{4,}"),
    "SRA":       re.compile(r"SR[APRX]\d{6,}"),
    "BioProject":re.compile(r"PRJNA?\d+"),
}


def find_dataset_urls(text: str) -> list[dict]:
    found, seen = [], set()
    for m in DOI_PATTERN.finditer(text):
        doi = m.group().rstrip(".")
        if doi not in seen:
            seen.add(doi); found.append({"type": "DOI", "value": doi})
    for name, pat in REPO_PATTERNS.items():
        for m in pat.finditer(text):
            u = m.group()
            if u not in seen:
                seen.add(u); found.append({"type": name, "value": u})
    for name in ["GEO", "SRA", "BioProject"]:
        for m in ACCESSION_PATTERNS[name].finditer(text):
            a = m.group()
            if a not in seen:
                seen.add(a); found.append({"type": name, "value": a})
    # Topic-specific patterns injected via update_patterns()
    for name, pat in _extra_url_patterns.items():
        for m in pat.finditer(text):
            v = m.group()
            if v not in seen:
                seen.add(v); found.append({"type": name, "value": v})
    return found


def find_data_availability_section(text: str) -> str:
    # Built-in academic headers (always active)
    headers = [
        r"data\s+availab", r"availability\s+of\s+data", r"data\s+access",
        r"code\s+availab", r"supplementary\s+(?:information|materials?|data)",
        r"supporting\s+information",
    ]
    # Merge LLM-injected topic-specific headers
    all_headers = headers + [h for h in _extra_section_headers if h not in headers]

    # Primary search: section that ends before references/acknowledgments
    pat = re.compile(
        r"(?:^|\n)\s*(?:" + "|".join(all_headers) + r").*?"
        r"(?=\n\s*(?:references|acknowledgment|author|conflict|funding|appendix)|\Z)",
        re.I | re.DOTALL,
    )
    m = pat.search(text)
    if m:
        return m.group().strip()[:2000]

    # Fix #18: Fallback — search from end of document for sections that appear
    # *after* references (common in many journals).
    tail_pat = re.compile(
        r"(?:^|\n)\s*(?:" + "|".join(all_headers) + r").*",
        re.I | re.DOTALL,
    )
    # Search in the last 20% of the document
    tail_start = max(0, int(len(text) * 0.80))
    tail_text = text[tail_start:]
    m2 = tail_pat.search(tail_text)
    return m2.group().strip()[:2000] if m2 else ""


# ── PDF table extraction ──────────────────────────────────────────────────────

def extract_tables_pdfplumber(pdf_path: str) -> list[list[list[str]]]:
    tables = []
    try:
        import pdfplumber
        with pdfplumber.open(pdf_path) as pdf:
            for pnum, page in enumerate(pdf.pages):
                for tbl in page.extract_tables() or []:
                    if tbl and len(tbl) > 1:
                        cleaned = [[(c.strip() if c else "") for c in row] for row in tbl]
                        tables.append(cleaned)
                        console.print(f"  [dim]Table on page {pnum+1}: {len(cleaned)}×{len(cleaned[0])}[/dim]")
    except ImportError:
        console.print("  [yellow]pdfplumber not installed — skipping table extraction[/yellow]")
    except Exception as e:
        console.print(f"  [yellow]Table extraction failed: {e}[/yellow]")
    return tables


def tables_to_csv_strings(tables: list[list[list[str]]]) -> list[str]:
    out = []
    for i, tbl in enumerate(tables):
        lines = [",".join(f'"{c}"' for c in row) for row in tbl]
        out.append(f"[Table {i+1}]\n" + "\n".join(lines))
    return out


def extract_tables_html(html: str) -> list[str]:
    try:
        soup = BeautifulSoup(html, "lxml")
        results = []
        for tbl in soup.find_all("table"):
            try:
                df = pd.read_html(io.StringIO(str(tbl)))[0]
                results.append(df.to_csv(index=False))
            except Exception:
                continue
        return results
    except Exception:
        return []


# ── OCR fallback ──────────────────────────────────────────────────────────────

def extract_pdf_text_ocr(pdf_path: str) -> str:
    if not cfg.ENABLE_OCR:
        return ""
    console.print("  [yellow]Starting OCR fallback…[/yellow]")
    parts = []
    try:
        images = convert_from_path(pdf_path)
        for i, img in enumerate(images):
            console.print(f"    [dim]OCR page {i+1}/{len(images)}[/dim]")
            arr = np.array(img)
            gray = cv2.cvtColor(arr, cv2.COLOR_RGB2GRAY)
            _, thresh = cv2.threshold(gray, 150, 255, cv2.THRESH_BINARY + cv2.THRESH_OTSU)
            parts.append(pytesseract.image_to_string(thresh))
    except Exception as e:
        console.print(f"  [red]OCR failed: {e}[/red]")
    return "\n\n".join(parts)


# ── Supplement links ──────────────────────────────────────────────────────────

def find_supplement_links(html: str, base_url: str) -> list[dict]:
    soup = BeautifulSoup(html, "lxml")
    results, seen = [], set()
    DATA_EXTS = {".csv", ".xlsx", ".xls", ".zip", ".tsv", ".json", ".xml"}
    SUP_KEYWORDS = {"supplement", "supporting", "additional", "appendix",
                    "data", "table s", "figure s", "dataset"}
    for a in soup.find_all("a", href=True):
        href = a["href"]
        text = a.get_text(strip=True).lower()
        ext = Path(href).suffix.lower()
        if (any(k in text for k in SUP_KEYWORDS) or ext in DATA_EXTS) and href not in seen:
            seen.add(href)
            if not href.startswith("http"):
                href = urljoin(base_url, href)
            results.append({"url": href, "filename": Path(href).name,
                             "type": ext, "text": a.get_text(strip=True)[:100]})
    return results


# ── Text cleaning and chunking ────────────────────────────────────────────────

def clean_text(text: str) -> str:
    # Collapse multi-line brackets
    text = re.sub(r'\[\s*\n\s*([A-Za-z0-9]+)\s*\n\s*\]', r'[\1]', text)
    text = re.sub(r'\[\s*\n\s*\]', '', text)
    # Strip inline citations
    text = re.sub(r'\[\s*\d+(?:[\s,\-]*\d+)*\s*\]', '', text)
    text = re.sub(r'\(\s*[A-Za-z]+ et al\.,?\s*\d{4}\s*\)', '', text)
    # Remove URLs and DOI noise
    text = re.sub(r'https?://\S+|www\.\S+', '', text)
    text = re.sub(r'\[\s*(?:DOI|PMC free article|PubMed|Google Scholar|Open in a new tab)\s*\]', '', text, flags=re.I)
    text = re.sub(r'doi:\s*10\.\S+', '', text, flags=re.I)
    # Remove sparse lines (standalone page/ref numbers)
    lines, cleaned = text.split('\n'), []
    for line in lines:
        s = line.strip()
        if len(s) < 3 and s.isdigit():
            continue
        cleaned.append(line)
    text = '\n'.join(cleaned)
    text = re.sub(r'\n{3,}', '\n\n', text)
    text = re.sub(r' {2,}', ' ', text)
    return text.strip()


def chunk_text(text: str, max_chars: int = cfg.MAX_TEXT_CHARS, overlap: int = 500) -> list[str]:
    """
    Section-aware text chunking for data extraction.

    Improvements over naive sliding window:
      1. Detects section headers and splits at section boundaries first
      2. Keeps tables and structured data blocks intact when possible
      3. Falls back to paragraph/sentence splitting within large sections
      4. Overlap is only applied at non-section boundaries to preserve context
    """
    text = clean_text(text)
    if len(text) <= max_chars:
        return [text]

    # Detect section boundaries: numbered sections, ALL CAPS headers, underlined text
    section_pattern = re.compile(
        r'\n(?='
        r'(?:\d{1,3}(?:\.\d{1,3})*\.?\s+[A-Z])'  # "1. Introduction", "2.3 Methods"
        r'|(?:[A-Z][A-Z\s]{4,}(?:\n|$))'           # "RESULTS AND DISCUSSION"
        r'|(?:[A-Z][a-z]+(?:\s+[A-Z][a-z]+)*\s*\n[-=]{3,})'  # Underlined headers
        r'|(?:#{1,4}\s+\S)'                         # Markdown headers
        r')'
    )

    # Split into sections at detected headers
    section_starts = [0] + [m.start() + 1 for m in section_pattern.finditer(text)]
    sections = []
    for i, start in enumerate(section_starts):
        end = section_starts[i + 1] if i + 1 < len(section_starts) else len(text)
        sec = text[start:end].strip()
        if sec:
            sections.append(sec)

    # Build chunks from sections, merging small sections, splitting large ones
    chunks: list[str] = []
    current_chunk = ""

    for section in sections:
        # If adding this section would exceed max_chars, finalize current chunk
        if current_chunk and len(current_chunk) + len(section) + 2 > max_chars:
            if current_chunk.strip():
                chunks.append(current_chunk.strip())
            current_chunk = ""

        # If a single section is too large, sub-split it
        if len(section) > max_chars:
            # First, finalize any accumulated content
            if current_chunk.strip():
                chunks.append(current_chunk.strip())
                current_chunk = ""

            # Sub-split large section at paragraph or sentence boundaries
            pos = 0
            while pos < len(section):
                end = pos + max_chars
                sub = section[pos:end]

                if end < len(section):
                    # Try to split at paragraph boundary
                    para_break = sub.rfind("\n\n")
                    if para_break > max_chars * 0.4:
                        sub = sub[:para_break]
                        end = pos + para_break
                    else:
                        # Fall back to sentence boundary
                        sent_break = sub.rfind(". ")
                        if sent_break > max_chars * 0.4:
                            sub = sub[:sent_break + 1]
                            end = pos + sent_break + 1

                if sub.strip():
                    chunks.append(sub.strip())

                # Add overlap for sub-splits within a section (continuation context)
                pos = end - overlap if end < len(section) else end
        else:
            # Section fits — accumulate into current chunk
            if current_chunk:
                current_chunk += "\n\n" + section
            else:
                current_chunk = section

    # Don't forget the last accumulated chunk
    if current_chunk.strip():
        chunks.append(current_chunk.strip())

    return chunks if chunks else [text[:max_chars]]

def _is_empty(v: str) -> bool:
    return str(v).strip().lower() in _EMPTY_VALS


def _split_col(col: str) -> tuple[str, str]:
    """Split 'Payload Drug (e.g. chemotherapy agent)' → ('Payload Drug', 'e.g. …')."""
    m = re.match(r'^(.*?)\s*[\(\[](e\.g\.?|i\.e\.?)(.+?)[\)\]]?\s*$', col, re.I)
    if m:
        hint = (m.group(2) + m.group(3)).rstrip(')].').strip()
        return m.group(1).strip(), hint
    return col.strip(), ""


def _dynamic_field_def(column_name: str, topic: str) -> str:
    """
    Generate a reasonable field definition from the column name and topic
    when no hardcoded definition exists.  Used as fallback for any topic.
    """
    key = column_name.strip().lower()
    # Generic patterns
    if re.search(r'\byear\b|\bdate\b', key):
        return f"4-digit year relevant to {topic} (e.g. 2021)."
    if re.search(r'\bname\b|\btitle\b', key):
        return f"Name or title of the {topic} item described in the text."
    if re.search(r'\bauthor\b|\bresearcher\b|\bscientist\b', key):
        return "Author or researcher names, comma-separated."
    if re.search(r'\bdoi\b|\bidentifier\b|\baccession\b', key):
        return "Unique identifier or DOI (e.g. 10.xxxx/xxxxx)."
    if re.search(r'\babstract\b|\bsummary\b|\bdescription\b', key):
        return "1-2 sentence summary of the item."
    if re.search(r'\bjournal\b|\bpublication\b|\bsource\b', key):
        return "Journal or publication name where this was published."
    if re.search(r'\bmethod\b|\btechnique\b|\bapproach\b|\bprotocol\b', key):
        return f"Experimental method or technique used in the {topic} study."
    if re.search(r'\bresult\b|\bfinding\b|\boutcome\b|\bconclusion\b', key):
        return f"Key result or finding reported for {topic}."
    if re.search(r'\bspecies\b|\borganism\b|\bstrain\b', key):
        return "Biological species or organism name (e.g. Homo sapiens, E. coli)."
    if re.search(r'\bcell\b|\btissue\b|\bsample\b', key):
        return "Cell type, tissue, or sample used in the experiment."
    if re.search(r'\bdose\b|\bconcentration\b|\bquantity\b|\bamount\b', key):
        return "Numeric dose or concentration with units (e.g. 10 mg/kg, 5 μM)."
    if re.search(r'\beffect\b|\bactivity\b|\bfunction\b', key):
        return f"Observed effect or activity related to {topic}."
    if re.search(r'\bgene\b|\bprotein\b|\bbiomarker\b|\bmarker\b', key):
        return "Gene symbol or protein name (e.g. BRCA1, p53, HER2)."
    if re.search(r'\bdisease\b|\bcondition\b|\bindication\b|\bpathology\b', key):
        return "Disease or medical condition name (e.g. lung cancer, diabetes)."
    if re.search(r'\bstatistic\b|\bpvalue\b|\bp.value\b|\bsignificance\b', key):
        return "Statistical measure or p-value (e.g. p < 0.05, OR = 2.3)."
    if re.search(r'\bcountry\b|\blocation\b|\bregion\b|\bsite\b', key):
        return "Geographic location or country name."
    # Generic fallback
    return (
        f'Extract the value for "{column_name}" from the text about {topic}. '
        "Use the exact value as written; do not paraphrase."
    )


def _build_field_block(columns: list[str], topic: str = "",
                       llm_defs: dict[str, str] | None = None) -> tuple[str, list[str], list[str]]:
    """
    Returns:
        field_block  – formatted string for prompt
        clean_keys   – list of field names without parentheticals
        hints        – list of raw hint strings
    """
    defs = llm_defs or _FIELD_DEFS
    clean_keys, hints = [], []
    lines = []
    for col in columns:
        key, hint = _split_col(col)
        clean_keys.append(key)
        hints.append(hint)
        # Try LLM-generated definition first; fall back to dynamic generator
        defn = next(
            (v for k, v in defs.items() if k in key.lower() or key.lower() in k),
            None,
        )
        if defn is None:
            defn = hint or _dynamic_field_def(key, topic)
        lines.append(f'  "{key}": {defn}')
    return "\n".join(lines), clean_keys, hints


def _build_example(clean_keys: list[str], topic: str) -> str:
    example = {}
    for key in clean_keys:
        val = next(
            (v for k, v in _FIELD_EXAMPLES.items() if k in key.lower() or key.lower() in k),
            "actual value from text",
        )
        example[key] = val
    return json.dumps([example], indent=2)


def _parse_json(text: str) -> list[dict] | None:
    """Robustly parse a JSON array or object from LLM response."""
    text = text.strip()

    # Strip preamble prose before first [ or {
    fb, bb = text.find('{'), text.find('[')
    starts = [i for i in [fb, bb] if i != -1]
    if starts:
        text = text[min(starts):]

    # Fix unquoted N/A
    text = re.sub(r':\s*N/A\s*([,}\]])', r': "N/A"\1', text)

    # Remove markdown fences
    text = re.sub(r'```(?:json)?\s*', '', text).strip().strip('`')

    for attempt in [text, text + ']', text + '}']:
        try:
            parsed = json.loads(attempt)
            if isinstance(parsed, list):
                return [d for d in parsed if isinstance(d, dict)]
            if isinstance(parsed, dict):
                return [parsed]
        except json.JSONDecodeError:
            pass

    # Last-ditch: find outermost array or object
    for bracket, closing in [('[', ']'), ('{', '}')]:
        si = text.find(bracket)
        if si == -1:
            continue
        ei = text.rfind(closing)
        if ei > si:
            try:
                parsed = json.loads(text[si:ei+1])
                if isinstance(parsed, list):
                    return [d for d in parsed if isinstance(d, dict)]
                if isinstance(parsed, dict):
                    return [parsed]
            except json.JSONDecodeError:
                pass

    return None


def _fuzzy_get(item_norm: dict, clean_key: str, full_col: str) -> str:
    """Look up a value trying clean key, full col, then substring match."""
    for key in [clean_key.lower(), full_col.lower(),
                re.sub(r'[\(\[\{].*?[\)\]\}]', '', full_col).strip().lower()]:
        v = item_norm.get(key)
        if v is not None:
            return str(v).strip()
    for k, v in item_norm.items():
        ck = clean_key.lower()
        if ck and (ck in k or k in ck):
            return str(v).strip()
    return "N/A"


def _get_embeddings(texts: list[str], model: str = "text-embedding-3-small") -> list[list[float]] | None:
    """
    Get embeddings from OpenAI API. Returns None on failure (triggers fallback).
    Truncates each text to ~8000 chars to stay within token limits.
    """
    try:
        import openai
        client = openai.OpenAI()
        truncated = [t[:8000] for t in texts]
        response = client.embeddings.create(input=truncated, model=model)
        return [item.embedding for item in response.data]
    except Exception as e:
        console.print(f"    [dim yellow]Embeddings unavailable ({e}), using structural scoring[/dim yellow]")
        return None


def _cosine_similarity(a: list[float], b: list[float]) -> float:
    """Compute cosine similarity between two vectors."""
    dot = sum(x * y for x, y in zip(a, b))
    norm_a = sum(x * x for x in a) ** 0.5
    norm_b = sum(x * x for x in b) ** 0.5
    if norm_a == 0 or norm_b == 0:
        return 0.0
    return dot / (norm_a * norm_b)


def _structural_bonus(chunk: str, chunk_idx: int, total_chunks: int) -> float:
    """
    Lightweight structural score (0.0 - 1.0) as a secondary signal.
    Captures formatting cues that embeddings can't see (tables, numbers, position).
    """
    cl = chunk.lower()
    raw_len = len(chunk.strip())
    score = 0.0

    # Hard disqualify junk sections
    first300 = cl[:300]
    if any(junk in first300 for junk in _JUNK_HEADERS):
        return -1.0  # sentinel: skip this chunk entirely

    # Table-like formatting
    table_sigs = ['||', '|', '\t', '---']
    if sum(1 for s in table_sigs if s in chunk) >= 2:
        score += 0.2

    # Numeric density
    numbers = len(re.findall(r'\b\d+(?:\.\d+)?\b', cl))
    if numbers > 5:
        score += min(numbers * 0.01, 0.15)

    # Key-value pairs
    kv = len(re.findall(r'(?:^|\n)\s*[A-Z][^:\n]{2,30}:\s*\S', chunk))
    if kv >= 2:
        score += 0.1

    # List items
    lists = len(re.findall(r'(?:^|\n)\s*(?:[-*]|\d{1,3}[.)]\s)', chunk))
    if lists >= 3:
        score += 0.1

    # Positional bonus
    if chunk_idx == 0:
        score += 0.15
    elif chunk_idx == total_chunks - 1 and total_chunks > 2:
        score += 0.05

    # Length penalty
    if raw_len < 150:
        score -= 0.3
    elif raw_len < 500:
        score -= 0.1

    return max(min(score, 1.0), 0.0)


def _select_best_chunks(chunks: list[str], columns: list[str], max_chunks: int) -> list[str]:
    """
    Adjacency-aware chunk selection optimised for data extraction.

    Key insight: adjacent chunks with similar content are continuations of the
    same data section (different rows from the same table). They must NOT be
    penalised. Only truly duplicated NON-adjacent chunks should be filtered.

    Algorithm:
      1. Embed chunks + query with text-embedding-3-small.
      2. Compute hybrid score = 0.65 * semantic + 0.35 * structural.
      3. Select all chunks above a dynamic relevance threshold.
      4. Apply adjacency boost: if chunk i is selected, boost chunks i-1 and i+1.
      5. Only penalise NON-adjacent chunks with >85% embedding similarity
         (true duplicates, not data continuations).
      6. If still over budget, keep highest-scoring; if under, fill from remaining.
      7. Return in original document order.

    Falls back to structural scoring if embeddings API is unavailable.
    """
    n = len(chunks)
    if n <= max_chunks:
        return chunks

    # Build the query from column names
    query = "Extract structured data for these fields: " + ", ".join(columns)

    # Compute structural bonuses for all chunks
    struct_scores = [_structural_bonus(chunks[i], i, n) for i in range(n)]

    # Try embeddings-based scoring
    all_texts = [query] + chunks
    embeddings = _get_embeddings(all_texts)

    if embeddings is not None:
        query_emb = embeddings[0]
        chunk_embs = embeddings[1:]

        semantic_scores = [_cosine_similarity(query_emb, ce) for ce in chunk_embs]

        # Hybrid scores: 65% semantic + 35% structural
        hybrid_scores: list[float] = []
        for i in range(n):
            if struct_scores[i] == -1.0:
                hybrid_scores.append(-1.0)
            else:
                hybrid_scores.append(0.65 * semantic_scores[i] + 0.35 * struct_scores[i])

        # Dynamic threshold: median of positive scores (adaptive to document quality)
        positive_scores = [s for s in hybrid_scores if s > 0]
        if positive_scores:
            sorted_pos = sorted(positive_scores)
            median = sorted_pos[len(sorted_pos) // 2]
            # Threshold = 70% of median — captures most data-bearing chunks
            threshold = median * 0.70
        else:
            threshold = 0.0

        # Phase 1: Select all chunks above threshold
        selected: set[int] = set()
        for i in range(n):
            if hybrid_scores[i] >= threshold and hybrid_scores[i] != -1.0:
                selected.add(i)

        # Always include chunk 0 (title/intro context)
        selected.add(0)

        # Phase 2: Adjacency boost — if chunk i is selected, also consider i-1 and i+1
        adjacency_candidates: set[int] = set()
        for i in selected.copy():
            for neighbor in [i - 1, i + 1]:
                if 0 <= neighbor < n and neighbor not in selected:
                    if hybrid_scores[neighbor] != -1.0:
                        # Boost: include neighbor if its score is at least 50% of threshold
                        if hybrid_scores[neighbor] >= threshold * 0.50:
                            adjacency_candidates.add(neighbor)

        selected.update(adjacency_candidates)

        # Phase 3: Remove only TRUE duplicates — non-adjacent chunks with >85% similarity
        if len(selected) > max_chunks:
            selected_list = sorted(selected)
            to_remove: set[int] = set()

            for a_pos, idx_a in enumerate(selected_list):
                if idx_a in to_remove:
                    continue
                for idx_b in selected_list[a_pos + 1:]:
                    if idx_b in to_remove:
                        continue
                    # Only penalise if chunks are NOT adjacent in the document
                    if abs(idx_a - idx_b) > 1:
                        sim = _cosine_similarity(chunk_embs[idx_a], chunk_embs[idx_b])
                        if sim > 0.85:
                            # Remove the lower-scoring duplicate
                            if hybrid_scores[idx_a] < hybrid_scores[idx_b]:
                                to_remove.add(idx_a)
                            else:
                                to_remove.add(idx_b)

            selected -= to_remove

        # Phase 4: If still over budget, trim lowest-scoring chunks
        if len(selected) > max_chunks:
            ranked = sorted(selected, key=lambda i: hybrid_scores[i], reverse=True)
            selected = set(ranked[:max_chunks])

        # Phase 5: If under budget, fill from best remaining chunks
        if len(selected) < max_chunks:
            remaining = [
                (i, hybrid_scores[i]) for i in range(n)
                if i not in selected and hybrid_scores[i] != -1.0
            ]
            remaining.sort(key=lambda x: x[1], reverse=True)
            for idx, _ in remaining:
                if len(selected) >= max_chunks:
                    break
                selected.add(idx)

        pos_scores = [s for s in semantic_scores if s > 0]
        if pos_scores:
            console.print(
                f"    [dim]Embedding scores: "
                f"min={min(pos_scores):.3f} "
                f"max={max(pos_scores):.3f} "
                f"threshold={threshold:.3f} "
                f"selected={len(selected)}/{n}[/dim]"
            )

    else:
        # Fallback: structural scoring with adjacency awareness
        selected = {0}  # always keep first chunk

        # Select all structurally relevant chunks
        for i in range(n):
            if struct_scores[i] > 0.1 and struct_scores[i] != -1.0:
                selected.add(i)

        # Adjacency boost for fallback too
        for i in list(selected):
            for neighbor in [i - 1, i + 1]:
                if 0 <= neighbor < n and struct_scores[neighbor] != -1.0:
                    selected.add(neighbor)

        # Trim if over budget
        if len(selected) > max_chunks:
            ranked = sorted(selected, key=lambda i: struct_scores[i], reverse=True)
            selected = set(ranked[:max_chunks])

    # Return in original document order
    ordered = sorted(selected)
    return [chunks[i] for i in ordered]
# ── LLM Client ────────────────────────────────────────────────────────────────

class LLMClient:
    """
    Unified LLM client.  Provider is resolved from config.LLM_PROVIDER.

    Extraction strategy:
      Claude / OpenAI  → large context, send full text, single pass.
                         Skip chunking entirely for docs < 180K chars.
      Ollama           → chunked extraction with parallel workers.
    """

    def __init__(self):
        self.provider = cfg.LLM_PROVIDER
        self.model    = cfg.active_model()
        self._client  = None
        # Cache for LLM-generated field examples — keyed by "topic|col1|col2|…"
        # so the expensive call is made once per unique (topic, columns) combination.
        self._example_cache: dict[str, str] = {}
        # Cache for LLM-generated field definitions per (topic, columns) combo.
        self._field_defs_cache: dict[str, dict[str, str]] = {}
        self._init()

    # ── Initialisation ──────────────────────────────────────────────────────

    def _init(self):
        if self.provider == "claude":
            if not cfg.ANTHROPIC_API_KEY or cfg.ANTHROPIC_API_KEY.startswith("sk-ant-replace"):
                raise ValueError(
                    "ANTHROPIC_API_KEY is not set. Add it to your .env file."
                )
            import anthropic
            self._client = anthropic.Anthropic(api_key=cfg.ANTHROPIC_API_KEY)
            console.print(f"[green]✓ Claude API ready — {self.model}[/green]")

        elif self.provider == "openai":
            if not cfg.OPENAI_API_KEY or cfg.OPENAI_API_KEY.startswith("sk-replace"):
                raise ValueError(
                    "OPENAI_API_KEY is not set. Add it to your .env file."
                )
            import openai
            self._client = openai.OpenAI(api_key=cfg.OPENAI_API_KEY)
            console.print(f"[green]✓ OpenAI API ready — {self.model}[/green]")

        else:  # ollama
            try:
                import ollama as _ollama
                models = _ollama.list()
                available = [m.model for m in models.models]
                base = self.model.split(":")[0]
                if not any(base in m for m in available):
                    console.print(
                        f"[red]✗ Ollama model '{self.model}' not found.[/red]\n"
                        f"  Available: {available}\n"
                        f"  Run: ollama pull {self.model}"
                    )
                    raise SystemExit(1)
                console.print(f"[green]✓ Ollama ready — {self.model}[/green]")
            except ImportError:
                raise ImportError("Install ollama: pip install ollama")

    # ── Low-level completion ────────────────────────────────────────────────

    def _complete(self, prompt: str, system: str = "", max_tokens: int = 4096) -> str:
        try:
            if self.provider == "claude":
                msgs = [{"role": "user", "content": prompt}]
                r = self._client.messages.create(
                    model=self.model,
                    max_tokens=max_tokens,
                    system=system,
                    messages=msgs,
                )
                return r.content[0].text.strip()

            elif self.provider == "openai":
                msgs = []
                if system:
                    msgs.append({"role": "system", "content": system})
                msgs.append({"role": "user", "content": prompt})
                r = self._client.chat.completions.create(
                    model=self.model,
                    max_tokens=max_tokens,
                    temperature=cfg.TEMPERATURE,
                    messages=msgs,
                )
                return r.choices[0].message.content.strip()

            else:  # ollama
                payload = {
                    "model": self.model,
                    "prompt": prompt,
                    "system": system,
                    "stream": False,
                    "options": {
                        "num_gpu": cfg.NUM_GPU_LAYERS,
                        "num_ctx": cfg.CONTEXT_WINDOW,
                        "temperature": cfg.TEMPERATURE,
                    },
                }
                resp = requests.post(
                    f"{cfg.OLLAMA_URL}/api/generate",
                    json=payload,
                    timeout=115,
                )
                resp.raise_for_status()
                return resp.json().get("response", "").strip()

        except Exception as e:
            console.print(f"  [yellow]⚠ LLM call failed: {e}[/yellow]")
            return ""

    # ── Search query generation ─────────────────────────────────────────────

    def generate_search_queries(self, topic: str) -> list[str]:
        """
        Ask the LLM to generate 4 search queries tailored to the topic.
        Works for any subject — academic, commercial, or general.
        """
        system = (
            "You are a research assistant. Generate exactly 4 web search queries "
            "to find high-quality, information-rich sources about the given topic. "
            "Queries should target different angles (overview, data, studies, lists). "
            "Do NOT hard-code any site: operators — keep queries generic. "
            "Return ONLY a JSON array of 4 query strings, nothing else."
        )
        raw = self._complete(
            f'Topic: "{topic}"\n\nGenerate 4 search queries as a JSON array:',
            system=system,
            max_tokens=256,
        )
        try:
            clean = re.sub(r'```(?:json)?\s*', '', raw).strip().strip('`')
            queries = json.loads(clean)
            if isinstance(queries, list) and queries:
                return [str(q) for q in queries[:4]]
        except Exception:
            pass
        # Fallback — generic but not topic-specific to avoid stale hardcoding
        return [
            f'{topic}',
            f'{topic} research paper review',
            f'{topic} data study results',
            f'{topic} article abstract doi',
        ]

    def generate_content_keywords(self, topic: str) -> list[str]:
        """
        Ask the LLM for 10-15 keywords that signal a rich, topic-relevant page
        (as opposed to a landing page, paywall screen, or ad-heavy blog).
        """
        system = (
            "You are a content analyst. Given a topic, return 10-15 lowercase keywords "
            "that would appear in a high-quality, information-dense article or data page "
            "about that topic (not just a homepage or advertisement). "
            "Return ONLY a JSON array of lowercase strings."
        )
        raw = self._complete(
            f'Topic: "{topic}"\n\nGenerate content-quality keywords as a JSON array:',
            system=system,
            max_tokens=256,
        )
        try:
            clean = re.sub(r'```(?:json)?\s*', '', raw).strip().strip('`')
            kws = json.loads(clean)
            if isinstance(kws, list) and kws:
                return [str(k).lower() for k in kws]
        except Exception:
            pass
        # Generic fallback — enough to catch actual content pages
        return [
            "result", "data", "analysis", "study", "method",
            "finding", "table", "figure", "abstract", "conclusion",
        ]

    # ── Field example generation ────────────────────────────────────────────

    def generate_field_examples(self, clean_keys: list[str], topic: str) -> str:
        """
        Ask the LLM for one realistic example row for the given columns and topic.
        Replaces the hardcoded ADC-specific _FIELD_EXAMPLES dict so non-ADC topics
        receive appropriate examples rather than "KADCYLA" or "HER2-positive cancer".
        Result is cached to avoid repeated API calls across chunks from the same document.
        """
        cache_key = f"{topic}|{'|'.join(clean_keys[:10])}"
        if cache_key in self._example_cache:
            return self._example_cache[cache_key]

        system = (
            "You are a data extraction expert. Given a topic and a list of column names, "
            "return exactly one realistic example JSON object showing what a correctly "
            "extracted row would look like. Use plausible real values — not descriptions "
            "or placeholders. Return ONLY a JSON array containing one object."
        )
        col_list = ", ".join(f'"{k}"' for k in clean_keys[:10])
        raw = self._complete(
            f'Topic: "{topic}"\nColumns: [{col_list}]\n\n'
            'Return one example row as: [{"col1": "realistic value", ...}]',
            system=system,
            max_tokens=512,
        )
        try:
            clean = re.sub(r'```(?:json)?\s*', '', raw).strip().strip('`')
            parsed = json.loads(clean)
            if isinstance(parsed, list) and parsed and isinstance(parsed[0], dict):
                result = json.dumps(parsed[:1], indent=2)
                self._example_cache[cache_key] = result
                return result
        except Exception:
            pass
        # Fallback to the static example builder — always returns something usable
        fallback = _build_example(clean_keys, topic)
        self._example_cache[cache_key] = fallback
        return fallback

            "Examples (vary by domain):\n"
            '  Science: ["Title", "Authors", "Year", "Method", "Key Finding", "DOI"]\n'
            '  Products: ["Product Name", "Brand", "Price", "Category", "Rating", "URL"]\n'
            '  Finance:  ["Company", "Ticker", "Revenue", "Year", "Sector", "Source"]'
        )
        raw = self._complete(
            f'Topic: "{topic}"\n\nSuggest columns as a JSON array:',
            system=system,
            max_tokens=512,
        )
        parsed = _parse_json(raw)
        if parsed and isinstance(parsed[0], dict):
            # sometimes returns [{"columns": [...]}]
            for v in parsed[0].values():
                if isinstance(v, list):
                    return [str(x) for x in v]
        if isinstance(parsed, list) and parsed and isinstance(parsed[0], str):
            return parsed  # type: ignore
        # flat list returned directly
        try:
            # Strip markdown fences before parsing
            clean = re.sub(r'```(?:json)?\s*', '', raw).strip().strip('`')
            lst = json.loads(clean)
            if isinstance(lst, list):
                return [str(x) for x in lst]
        except Exception:
            pass
        # fallback: numbered list
        cols = []
        for line in raw.splitlines():
            m = re.match(r'^[\d\.\)\-\*\s]*(.+)$', line.strip())
            if m:
                c = m.group(1).strip().strip('"').strip("'")
                if c and len(c) < 80:
                    cols.append(c)
        return cols or ["Title", "Authors", "Year", "Abstract"]

    # ── Main extraction entry point ─────────────────────────────────────────

    def extract(
        self,
        text: str,
        columns: list[str],
        topic: str = "",
        is_table: bool = False,
        source_url: str = "",
        pre_chunked: list[str] | None = None,
    ) -> list[dict]:
        """
        Extract structured rows from text.

        Claude/OpenAI: sends full text (up to 180K chars) in one shot.
        Ollama:        chunked extraction with smart merge.

        pre_chunked – if provided (Ollama only), use these chunks directly
                      instead of re-chunking the text, preventing double-chunking
                      with different settings (Fix #6).
        """
        # Use LLM-generated field definitions (cached per topic+columns)
        llm_defs = self.generate_field_definitions(columns, topic)
        field_block, clean_keys, _ = _build_field_block(columns, topic, llm_defs=llm_defs)
        # generate_field_examples uses the LLM to build a topic-appropriate example row
        # and caches the result so subsequent documents don't pay the API cost again.
        example_json = self.generate_field_examples(clean_keys, topic)

        if self.provider in ("claude", "openai"):
            rows = self._extract_api(text, columns, clean_keys, field_block, example_json, topic, is_table)
        else:
            rows = self._extract_ollama(
                text, columns, clean_keys, field_block, example_json, topic, is_table,
                pre_chunked=pre_chunked,
            )

        # Attach source URL
        if source_url:
            for r in rows:
                r["Source_URL"] = source_url
        return rows

    # ── API extraction (Claude / OpenAI) ────────────────────────────────────

    def _build_column_rules(self, columns: list[str], topic: str) -> str:
        """
        Dynamically build per-column extraction rules from the actual column names.
        This replaces the hardcoded ADC-specific rules so any topic works correctly.
        Rules are numbered starting at 4 (rules 1-3 are universal, set in caller).
        """
        rules: list[str] = []
        rulenum = 4
        for col in columns:
            key, _ = _split_col(col)
            # Look up definition from static cache first, then dynamic generator
            defn = next(
                (v for k, v in _FIELD_DEFS.items() if k in key.lower() or key.lower() in k),
                None,
            )
            if defn is None:
                defn = _dynamic_field_def(key, topic)
            # Trim to a single-line rule (no walls of text in the system prompt)
            brief = defn.split(".")[0].rstrip()
            rules.append(f"{rulenum}. {key}: {brief}.")
            rulenum += 1
            if rulenum >= 10:          # cap at 6 column-specific rules
                break
        return "\n".join(rules) + "\n" if rules else ""

    def _extraction_system(self, field_block: str, example_json: str, topic: str,
                            is_table: bool, columns: list[str] | None = None) -> str:
        entity_rule = (
            "This text is a DATA TABLE — return ONE JSON object per data row."
            if is_table else
            "If the text lists multiple distinct items (drugs, papers, compounds, products…), "
            "return one object per item. If it discusses a single topic, return one object."
        )
        col_rules = self._build_column_rules(columns or [], topic)
        last_rule  = 4 + len([c for c in (columns or []) if len(col_rules) > 0])
        return (
            f'You are a data extraction expert. Topic: "{topic}".\n\n'
            "TASK: Extract structured data and return ONLY a valid JSON array.\n\n"
            f"FIELD DEFINITIONS (read carefully):\n{field_block}\n\n"
            f"CORRECT OUTPUT EXAMPLE:\n{example_json}\n\n"
            "STRICT RULES:\n"
            "1. Return ONLY a JSON array — zero prose, no markdown fences.\n"
            '2. Use exactly the field names shown. Use "N/A" when data is absent.\n'
            "3. NEVER copy definition text as a value — extract the real value from the text.\n"
            + col_rules +
            "Keep values concise (< 15 words per field).\n"
            f"{entity_rule}"
        )

    def _extract_api(
        self,
        text: str,
        columns: list[str],
        clean_keys: list[str],
        field_block: str,
        example_json: str,
        topic: str,
        is_table: bool,
    ) -> list[dict]:
        """Single-pass extraction for large-context APIs."""
        MAX_CHARS = 180_000
        if len(text) > MAX_CHARS:
            # For very long docs, chunk but use large chunks
            return self._extract_chunked_api(
                text, columns, clean_keys, field_block, example_json, topic, is_table, MAX_CHARS
            )

        system = self._extraction_system(field_block, example_json, topic, is_table, columns)
        raw = self._complete(text, system=system, max_tokens=8192)
        rows = self._parse_and_normalise(raw, columns, clean_keys)
        return self._filter_rows(rows, columns)

    def _extract_chunked_api(
        self,
        text: str,
        columns: list[str],
        clean_keys: list[str],
        field_block: str,
        example_json: str,
        topic: str,
        is_table: bool,
        chunk_size: int,
    ) -> list[dict]:
        from extractor import chunk_text
        chunks = chunk_text(text, max_chars=chunk_size, overlap=1000)
        system = self._extraction_system(field_block, example_json, topic, is_table, columns)

        all_rows: list[dict] = []
        for i, chunk in enumerate(chunks):
            console.print(f"    [dim]API chunk {i+1}/{len(chunks)}…[/dim]")
            raw = self._complete(chunk, system=system, max_tokens=8192)
            rows = self._parse_and_normalise(raw, columns, clean_keys)
            all_rows.extend(rows)

        return merge_chunk_results(all_rows, columns)

    # ── Ollama chunked extraction ────────────────────────────────────────────

    # Ollama config: how many top-scored chunks to send per document
    _OLLAMA_MAX_CHUNKS = int(os.getenv("OLLAMA_MAX_CHUNKS", "6"))

    def _extract_ollama(
        self,
        text: str,
        columns: list[str],
        clean_keys: list[str],
        field_block: str,
        example_json: str,
        topic: str,
        is_table: bool,
        pre_chunked: list[str] | None = None,
    ) -> list[dict]:
        from extractor import chunk_text, clean_text

        if pre_chunked is not None:
            # Use caller-supplied chunks directly — avoids double-chunking with
            # different max_chars/overlap settings (Fix #6).
            all_chunks = pre_chunked
        else:
            # Use larger chunks to reduce total count (fewer LLM calls = fewer timeouts)
            CHUNK_SIZE = min(cfg.MAX_TEXT_CHARS * 2, 20000)
            text = clean_text(text)
            all_chunks = chunk_text(text, max_chars=CHUNK_SIZE, overlap=300)

        # Score every chunk and pick the top N most data-dense ones.
        # This is the key fix: a 200K paper produces ~12 large chunks;
        # we send only the 6 highest-scoring ones instead of all 33 small ones.
        selected = _select_best_chunks(all_chunks, columns, self._OLLAMA_MAX_CHUNKS)
        console.print(
            f"    [dim]Selected {len(selected)}/{len(all_chunks)} chunks "
            f"(scored, top {self._OLLAMA_MAX_CHUNKS})[/dim]"
        )

        system = self._extraction_system(field_block, example_json, topic, is_table, columns)  # slightly under Ollama's default 120s read timeout

        # SEQUENTIAL processing — Ollama is single-threaded on one GPU/CPU.
        # Parallel calls just queue up and time out; sequential is faster overall.
        all_raw: list[list[dict]] = []
        for i, chunk in enumerate(selected):
            prompt = (
                f"FIELD DEFINITIONS:\n{field_block}\n\n"
                f"EXAMPLE OUTPUT:\n{example_json}\n\n"
                f"TEXT:\n---\n{chunk}\n---\n\n"
                "Return JSON array only. Do not write any explanation:"
            )
            console.print(f"    [dim]Chunk {i+1}/{len(selected)} ({len(chunk):,} chars)…[/dim]")
            try:
                raw = self._complete(prompt, system=system, max_tokens=1024)
                rows = self._parse_and_normalise(raw, columns, clean_keys)
                all_raw.append(rows)
                console.print(f"    [dim]  → {len(rows)} row(s)[/dim]")
            except Exception as e:
                console.print(f"    [yellow]⚠ chunk {i+1} failed: {e}[/yellow]")

        flat = [row for chunk_rows in all_raw for row in chunk_rows]
        merged = merge_chunk_results(flat, columns)
        return self._filter_rows(merged, columns)

    # ── Parsing helpers ─────────────────────────────────────────────────────

    def _parse_and_normalise(self, raw: str, columns: list[str], clean_keys: list[str]) -> list[dict]:
        parsed = _parse_json(raw)
        if not parsed:
            if cfg.DEBUG:
                console.print(f"    [dim red]Failed to parse: {raw[:200]}[/dim red]")
            return []
        results = []
        for item in parsed:
            if not isinstance(item, dict):
                continue
            norm = {k.strip().lower(): str(v).strip() for k, v in item.items()}
            row = {col: _fuzzy_get(norm, ck, col) for col, ck in zip(columns, clean_keys)}
            results.append(row)
        return results

    def _filter_rows(self, rows: list[dict], columns: list[str]) -> list[dict]:
        """Drop rows where >65% of fields are N/A."""
        min_valid = max(1, int(len(columns) * 0.35))
        good = [
            r for r in rows
            if sum(1 for c in columns if not _is_empty(r.get(c, "N/A"))) >= min_valid
        ]
        return good or rows[:1]  # always return at least the best row


# ── Smart merge ───────────────────────────────────────────────────────────────

def merge_chunk_results(rows: list[dict], columns: list[str]) -> list[dict]:
    """
    Merge extraction results intelligently.

    - If user selected a name/identity column (Trade Name, Generic Name, Title…)
      group rows by that anchor and merge same-entity results from different chunks.
    - If only technical columns (Target, Payload, Linker…) were selected,
      keep every row separate — merging them would produce one giant cell.
    - Deduplicates exact duplicate rows in both cases.
    """
    if not rows:
        return []

    col_lowers = [c.lower() for c in columns]

    # Normalise keys
    norm = [{k.strip().lower(): str(v).strip() for k, v in r.items()} for r in rows]

    # Find anchor column — use LLM-generated merge config if available,
    # otherwise fall back to generic heuristic keywords
    _NAME_KW   = _merge_identity_kw or {"name", "title", "product", "brand", "company",
                   "drug", "compound", "item", "entity", "subject", "movie", "film",
                   "song", "book", "stock", "ticker", "person"}
    _ANTI_KW   = _merge_content_kw or {"description", "summary", "detail", "note",
                   "comment", "year", "date", "doi", "authors", "abstract",
                   "source", "url", "link", "rating", "score", "price"}

    anchor_col: str | None = None
    for cl in col_lowers:
        if any(kw in cl for kw in _NAME_KW) and not any(ak in cl for ak in _ANTI_KW):
            anchor_col = cl
            break

    # Check if anchor has useful values (>50% populated)
    use_anchor = False
    if anchor_col:
        anchor_vals = [_get_val(item, anchor_col) for item in norm]
        filled = sum(1 for v in anchor_vals if v != "N/A")
        use_anchor = filled / max(len(anchor_vals), 1) >= 0.5

    # Group
    groups: dict[Any, list[dict]] = defaultdict(list)
    if use_anchor:
        for item, av in zip(norm, anchor_vals):
            key = av.lower() if av != "N/A" else id(item)
            groups[key].append(item)
    else:
        # No identity anchor — keep EVERY row fully separate.
        # Using a sequential counter guarantees no two rows share a key,
        # preventing the inadvertent merging of distinct multi-entity rows
        # (e.g. 10 drugs in a table being collapsed into 1 semicolon-joined row).
        for idx, item in enumerate(norm):
            groups[idx].append(item)

    # Merge each group
    merged: list[dict] = []
    for items in groups.values():
        row: dict = {}
        for col, cl in zip(columns, col_lowers):
            is_id = any(kw in cl for kw in _NAME_KW) and not any(ak in cl for ak in _ANTI_KW)
            vals = [_get_val(item, cl) for item in items if _get_val(item, cl) != "N/A"]
            if not vals:
                row[col] = "N/A"
            elif is_id:
                row[col] = max(vals, key=len)
            else:
                seen: set[str] = set()
                parts: list[str] = []
                for v in vals:
                    for p in re.split(r";\s*", v):
                        pk = p.strip().lower()
                        if p.strip() and pk not in seen and not _is_empty(pk):
                            seen.add(pk)
                            parts.append(p.strip())
                row[col] = "; ".join(parts[:12]) if parts else "N/A"
        merged.append(row)

    # Dedup exact rows
    seen_sigs: set[str] = set()
    deduped: list[dict] = []
    for row in merged:
        sig = "|".join(str(row.get(c, "")).lower() for c in columns)
        if sig not in seen_sigs:
            seen_sigs.add(sig)
            deduped.append(row)

    # Drop near-empty rows
    min_v = max(1, len(columns) // 4)
    filtered = [r for r in deduped if sum(1 for c in columns if not _is_empty(r.get(c, "N/A"))) >= min_v]
    return filtered or deduped[:1]


def _get_val(d: dict, col: str) -> str:
    raw = str(d.get(col, "N/A")).strip()
    if _is_empty(raw):
        # Try stripping parentheticals
        clean = re.sub(r'[\(\[\{].*?[\)\]\}]', '', col).strip()
        if clean != col:
            raw = str(d.get(clean, "N/A")).strip()
    return "N/A" if _is_empty(raw) else raw

        for i, col in enumerate(suggested, 1):

        sel = Prompt.ask(

                       if x.strip().isdigit() and 0 < int(x.strip()) <= len(suggested)]

# Minimum characters to consider a source as "full text" not just an abstract.
# A typical abstract is 150-300 words (~1000 chars). Full papers are 5000+ chars.
_MIN_FULLTEXT_CHARS = 3000

# Max scrape attempts per URL before giving up
_MAX_SCRAPE_ATTEMPTS = 2

# Reasons a source gets skipped — logged clearly for the user
_SKIP_REASONS = {
    "no_text":       "No text could be extracted (blocked or empty)",
    "abstract_only": "Too short — abstract only, not full text",
    "low_quality":   "Content too thin to extract meaningful data",
}

# LLM-generated topic-specific content keywords — set once per run by run_pipeline.
# These supplement the universal defaults to catch domain-specific content pages.
_topic_content_keywords: list[str] = []

# Universal keywords that indicate a real content page (not a landing/paywall page)
_DEFAULT_CONTENT_KEYWORDS: list[str] = [
    "method", "material", "experiment", "protocol", "procedure",
    "result", "finding", "conclusion", "discussion", "abstract",
    "figure", "table", "data", "analysis", "study",
]


def _check_content_quality(text: str, method: str) -> tuple[bool, str]:
    """
    Returns (is_good, reason).
    Checks if scraped content is rich enough to extract data from.
    Uses both universal defaults and LLM-generated topic keywords.
    """
    length = len(text.strip())

    # Hard minimum — anything shorter is abstract/snippet only
    if length < _MIN_FULLTEXT_CHARS:
        return False, f"abstract_only ({length:,} chars, need {_MIN_FULLTEXT_CHARS:,}+)"

    # HTML-scraped content with no meaningful section keywords = landing page
    if method == "html":
        text_lower = text.lower()
        all_keywords = _DEFAULT_CONTENT_KEYWORDS + _topic_content_keywords
        has_content = any(w in text_lower for w in all_keywords)
        if not has_content:
            return False, "landing_page (no topic-relevant content detected)"

    return True, "ok"


def _scrape_with_retry(source: str, is_local_pdf: bool) -> dict | None:
    """
    Attempt to scrape a source with exponential backoff retries (Fix #7).
    For URLs: retries up to _MAX_SCRAPE_ATTEMPTS times, doubling the wait each time.
    For local PDFs: no retry needed (filesystem is deterministic).
    """
    if is_local_pdf:
        return scrape_local_pdf(source)

    for attempt in range(1, _MAX_SCRAPE_ATTEMPTS + 1):
        scraped = scrape_url(source)
        if scraped and scraped.get("text"):
            return scraped
        if attempt < _MAX_SCRAPE_ATTEMPTS:
            wait = 2 ** attempt  # 2s, 4s
            console.print(
                f"  [yellow]⚠ Scrape attempt {attempt}/{_MAX_SCRAPE_ATTEMPTS} failed "
                f"— retrying in {wait}s…[/yellow]"
            )
            time.sleep(wait)
        else:
            console.print(
                f"  [red]✗ All {_MAX_SCRAPE_ATTEMPTS} scrape attempts failed for {source[:60]}[/red]"
            )
    return None

# ── Single-source processor ───────────────────────────────────────────────────

def process_source(
    source: str,
    columns: list[str],
    topic: str,
    client: LLMClient,
    is_local_pdf: bool = False,
) -> list[dict]:
    """
    Process one source (URL or local PDF path) and return extracted rows.
    source     – URL string or local file path
    columns    – user-selected field names
    topic      – original research topic (for context)
    client     – LLMClient instance
    is_local_pdf – True when source is a local .pdf file
    """
    # 1. Scrape (with retry for URLs)
    scraped = _scrape_with_retry(source, is_local_pdf)
    if not scraped or not scraped.get("text"):
        console.print(f"  [red]✗ SKIPPED — No text extracted from {str(source)[:60]}[/red]")
        return []

    full_text = scraped["text"]
    method    = scraped.get("method", "unknown")

    # ── Quality gate: skip abstract-only or blocked pages ────────────────────
    is_good, reason = _check_content_quality(full_text, method)
    if not is_good:
        console.print(f"  [yellow]⚠ Abstract-only or thin content — {reason}[/yellow]")
        console.print(f"  [dim]  Source: {str(source)[:70]}[/dim]")
        # ── Open-access fallback: look for a free full-text version ──────────
        if not is_local_pdf:
            fallback_urls = find_fulltext_fallback(source)
            for fb_url in fallback_urls:
                console.print(f"  [cyan]↩ Trying fallback: {fb_url[:80]}[/cyan]")
                fb_scraped = _scrape_with_retry(fb_url, is_local_pdf=False)
                if not fb_scraped or not fb_scraped.get("text"):
                    continue
                fb_good, fb_reason = _check_content_quality(
                    fb_scraped["text"], fb_scraped.get("method", "unknown")
                )
                if fb_good:
                    console.print(
                        f"  [green]✓ Fallback succeeded ({len(fb_scraped['text']):,} chars "
                        f"via {fb_scraped.get('method')})[/green]"
                    )
                    scraped = fb_scraped
                    full_text = fb_scraped["text"]
                    method    = fb_scraped.get("method", "unknown")
                    source    = fb_url          # update source URL for attribution
                    break
                else:
                    console.print(f"  [dim]  Fallback also thin: {fb_reason}[/dim]")
            else:
                # all fallbacks exhausted (or none found)
                console.print(f"  [red]✗ SKIPPED — no open full-text found[/red]")
                return []
        else:
            return []

    console.print(f"  [dim]✓ Quality check passed ({len(full_text):,} chars via {method})[/dim]")
    pdf_path  = scraped.get("pdf_path")
    html      = scraped.get("html", "")

    # ── Derive a safe stem name for output files ──────────────────────────
    if is_local_pdf:
        stem = Path(source).stem
    else:
        stem = re.sub(r"[^\w\-]", "_", source.split("//")[-1])[:60]

    # ── Save text chunks ──────────────────────────────────────────────────
    chunks = chunk_text(full_text)
    chunks_dir = Path(cfg.CHUNK_TEMP_DIR)
    chunks_dir.mkdir(parents=True, exist_ok=True)
    for idx, chunk in enumerate(chunks):
        chunk_file = chunks_dir / f"{stem}_chunk{idx+1:03d}.txt"
        chunk_file.write_text(chunk, encoding="utf-8")
    console.print(f"  [dim]Saved {len(chunks)} text chunk(s) → {cfg.CHUNK_TEMP_DIR}[/dim]")
    # ── Extract and save images from PDF (URL-scraped PDFs only;
    #    local PDFs are handled inside scrape_local_pdf) ───────────────────
    if pdf_path and not is_local_pdf:
        images_dir = str(Path("data/images") / stem)
        n_imgs = extract_images_from_pdf(pdf_path, images_dir)
        if n_imgs:
            console.print(f"  [dim]Saved {n_imgs} image(s) → {images_dir}[/dim]")


    # 2. Collect table chunks (tables carry the richest structured data)
    # Fix #3/#17: Filter out tables with fewer than 2 columns OR fewer than 2 data rows
    # to avoid wasting LLM calls on near-empty pdfplumber artefacts.
    table_chunks: list[str] = []
    tables_dir = Path("data/tables") / stem
    if pdf_path:
        raw_tables = extract_tables_pdfplumber(pdf_path)
        # Filter: require ≥2 columns and ≥2 rows (header + at least one data row)
        tables = [t for t in raw_tables if len(t) >= 2 and len(t[0]) >= 2]
        if len(tables) < len(raw_tables):
            console.print(
                f"  [dim]Filtered {len(raw_tables) - len(tables)} narrow/empty table(s) "
                f"(kept {len(tables)})[/dim]"
            )
        if tables:
            table_chunks = tables_to_csv_strings(tables)
            tables_dir.mkdir(parents=True, exist_ok=True)
            for i, tbl in enumerate(tables):
                tbl_path = tables_dir / f"table_{i+1:03d}.csv"
                rows_csv = "\n".join(",".join(f'"{c}"' for c in row) for row in tbl)
                tbl_path.write_text(rows_csv, encoding="utf-8")
            console.print(f"  [dim]Found {len(tables)} PDF table(s) → {tables_dir}[/dim]")
    if html:
        html_tables = extract_tables_html(html)
        if html_tables:
            table_chunks.extend(html_tables)
            tables_dir.mkdir(parents=True, exist_ok=True)
            for i, csv_str in enumerate(html_tables):
                tbl_path = tables_dir / f"html_table_{i+1:03d}.csv"
                tbl_path.write_text(csv_str, encoding="utf-8")
            console.print(f"  [dim]Found {len(html_tables)} HTML table(s) → {tables_dir}[/dim]")

    # 3. Build extra metadata context
    dataset_links = find_dataset_urls(full_text)
    data_avail    = find_data_availability_section(full_text)
    supplements   = find_supplement_links(html, source) if html else []

    # ── Save supplement links ─────────────────────────────────────────────
    if supplements:
        sup_dir = Path(cfg.SUPPLEMENT_TEMP_DIR)
        sup_dir.mkdir(parents=True, exist_ok=True)
        sup_file = sup_dir / f"{stem}_supplements.json"
        sup_file.write_text(json.dumps(supplements, indent=2), encoding="utf-8")
        console.print(f"  [dim]Saved {len(supplements)} supplement link(s) → {sup_file}[/dim]")

    extra: list[str] = []
    if data_avail:
        extra.append(f"[DATA AVAILABILITY]\n{data_avail}")
    if dataset_links:
        extra.append("[DATASET LINKS]\n" + "; ".join(f"{d['type']}: {d['value']}" for d in dataset_links[:5]))
    if supplements:
        extra.append("[SUPPLEMENTARY FILES]\n" + ", ".join(s["filename"] for s in supplements[:5]))
    extra_context = "\n\n".join(extra)

    llm_columns = [c for c in columns if c not in ("Source_URL", "Dataset_Links")]

    all_rows: list[dict] = []

    # Fix #6: For Ollama, pass the already-saved chunks so _extract_ollama
    # does not re-chunk with different settings.
    pre_chunked = chunks if client.provider == "ollama" else None

    # 4. Extract from main text
    text_with_ctx = full_text + ("\n\n" + extra_context if extra_context else "")
    console.print(f"  [dim]Extracting from text ({len(text_with_ctx):,} chars)…[/dim]")
    rows = client.extract(
        text_with_ctx, llm_columns, topic=topic, is_table=False,
        source_url=source, pre_chunked=pre_chunked,
    )
    all_rows.extend(rows)
    console.print(f"  [dim]Text pass → {len(rows)} row(s)[/dim]")

    # 5. Extract from table chunks (each is a separate pass — tables often
    #    have the clearest structured data)
    for i, tbl in enumerate(table_chunks[:5]):
        console.print(f"  [dim]Extracting from table {i+1}/{min(len(table_chunks), 5)}…[/dim]")
        tbl_rows = client.extract(
            f"--- DATA TABLE {i+1} ---\n{tbl}",
            llm_columns, topic=topic, is_table=True, source_url=source,
        )
        console.print(f"  [dim]Table {i+1} → {len(tbl_rows)} row(s)[/dim]")
        all_rows.extend(tbl_rows)

    # 6. Merge across all passes
    merged = merge_chunk_results(all_rows, llm_columns)

    # Attach metadata
    for row in merged:
        row["Source_URL"] = source
        if dataset_links:
            row["Dataset_Links"] = "; ".join(f"{d['type']}: {d['value']}" for d in dataset_links[:5])

    console.print(f"  [bold green]✓ {len(merged)} row(s) extracted[/bold green]")
    return merged


def _save_partial(rows: list[dict], columns: list[str], output_path: str) -> None:
    """Save partial results to CSV after each successful source (Fix #11)."""
    if not rows:
        return
    try:
        df = pd.DataFrame(rows)
        for col in columns:
            if col not in df.columns:
                df[col] = "N/A"
        extra_cols = [c for c in df.columns if c not in columns]
        df = df[columns + extra_cols]
        os.makedirs(os.path.dirname(output_path) or ".", exist_ok=True)
        df.to_csv(output_path, index=False, encoding="utf-8-sig")
    except Exception as e:
        console.print(f"  [yellow]⚠ Partial save failed: {e}[/yellow]")


# ── Main pipeline ─────────────────────────────────────────────────────────────

def run_pipeline(
    topic: str,
    columns: list[str],
    client: LLMClient,
    sources: list[str] | None = None,
    local_pdfs: list[str] | None = None,
    limit: int = cfg.DEFAULT_LIMIT,
    output_path: str = cfg.OUTPUT_FILE,
) -> pd.DataFrame:
    """
    Main pipeline entry point.

    sources    – list of URLs (or None → search DuckDuckGo)
    local_pdfs – list of local PDF paths (or None)
    """
    all_rows: list[dict] = []
    failed = 0

    # ── Generate topic-specific domain & paywall config via LLM ─────────────
    console.print("[dim]Generating topic-specific domain configuration…[/dim]")
    domain_config = client.generate_domain_config(topic)
    configure_for_topic(domain_config)
    configure_paywall_detection(
        domain_config.get("paywall_domains", []),
        domain_config.get("paywall_signals", []),
    )

    # ── Generate topic-specific keywords and search queries via LLM ──────────
    global _topic_content_keywords
    console.print("[dim]Generating topic-specific content keywords…[/dim]")
    _topic_content_keywords = client.generate_content_keywords(topic)

    # ── Generate merge config (identity vs content columns) ────────────────
    console.print("[dim]Generating merge configuration…[/dim]")
    client.generate_merge_config(columns, topic)

    # ── Inject topic-specific patterns into extractor (section headers + URL patterns)
    console.print("[dim]Generating topic-specific extraction patterns…[/dim]")
    import extractor as _extractor
    _extractor.update_patterns(
        section_headers=client.generate_source_section_headers(topic),
        url_patterns=client.generate_url_patterns(topic),
    )

    # ── Mode A: local PDFs ───────────────────────────────────────────────────
    if local_pdfs:
        if len(local_pdfs) > limit:
            console.print(f"[dim]Limiting to {limit} of {len(local_pdfs)} PDFs (use --limit to change)[/dim]")
            local_pdfs = local_pdfs[:limit]
        console.print(f"\n[bold]Processing {len(local_pdfs)} local PDF(s)…[/bold]")
        with Progress(SpinnerColumn(), TextColumn("{task.description}"),
                      BarColumn(), TextColumn("{task.completed}/{task.total}"),
                      console=console) as prog:
            task = prog.add_task("PDFs", total=len(local_pdfs))
            for path in local_pdfs:
                console.print(f"\n[cyan]► {Path(path).name}[/cyan]")
                try:
                    rows = process_source(path, columns, topic, client, is_local_pdf=True)
                    if rows:
                        all_rows.extend(rows)
                        # Fix #11: partial save after every successful source
                        _save_partial(all_rows, columns, output_path)
                        prog.update(task, advance=1, description=f"✓ {len(all_rows)} rows")
                    else:
                        failed += 1
                        prog.update(task, advance=1, description=f"✗ failed: {Path(path).name}")
                except Exception as e:
                    failed += 1
                    console.print(f"  [red]✗ Error: {e}[/red]")
                    prog.update(task, advance=1)

    # ── Mode B: URLs (provided or searched) ─────────────────────────────────
    else:
        if not sources:
            console.print(f"\n[bold]Searching for '{topic}'…[/bold]")
            console.print("[dim]Generating topic-specific search queries…[/dim]")
            search_queries = client.generate_search_queries(topic)
            # Fetch 3x the limit so we have spares to replace skipped sources
            results = search_sources(topic, limit=limit * 3, custom_queries=search_queries)
            sources = [r["url"] for r in results]
            if not sources:
                console.print("[red]No search results found.[/red]")
                return pd.DataFrame()
            console.print(f"[green]✓ Found {len(sources)} candidate sources (targeting {limit} good ones)[/green]")

        try:
            init_browser()
        except Exception as e:
            console.print(f"[yellow]Browser init failed ({e}), using requests fallback[/yellow]")

        good_count  = 0   # sources that passed quality check
        skipped     = 0   # sources that were skipped (abstract-only, blocked)

        with Progress(SpinnerColumn(), TextColumn("{task.description}"),
                      BarColumn(), TextColumn("{task.completed}/{task.total}"),
                      console=console) as prog:
            task = prog.add_task("Sources", total=limit)
            for url in sources:
                if good_count >= limit:
                    break  # we have enough good sources
                console.print(f"\n[cyan]► {url[:80]}[/cyan]")
                try:
                    rows = process_source(url, columns, topic, client, is_local_pdf=False)
                    if rows:
                        all_rows.extend(rows)
                        good_count += 1
                        # Fix #11: partial save after every successful source
                        _save_partial(all_rows, columns, output_path)
                        prog.update(task, advance=1, description=f"✓ {len(all_rows)} rows ({good_count}/{limit} good)")
                    else:
                        skipped += 1
                        console.print(f"  [dim]Skipped {skipped} source(s) so far — trying next…[/dim]")
                except Exception as e:
                    skipped += 1
                    console.print(f"  [red]✗ Error: {e}[/red]")

        failed = skipped
        console.print(
            f"\n[dim]Summary: {good_count} good sources, {skipped} skipped "
            f"(abstract-only / blocked / error)[/dim]"
        )

        close_browser()

    if not all_rows:
        console.print("[red]No data extracted.[/red]")
        return pd.DataFrame()

    # Fix #8: Global deduplication across all sources before building final DataFrame.
    # Preserve Source_URL and Dataset_Links BEFORE merging (merge only works on
    # llm_columns and silently drops everything else).
    llm_columns = [c for c in columns if c not in ("Source_URL", "Dataset_Links")]

    # Save metadata keyed by row index so we can restore it after dedup
    pre_merge_meta = [
        {
            "Source_URL":    row.get("Source_URL", ""),
            "Dataset_Links": row.get("Dataset_Links", ""),
        }
        for row in all_rows
    ]

    console.print(f"[dim]Deduplicating {len(all_rows)} total rows across all sources…[/dim]")
    all_rows = merge_chunk_results(all_rows, llm_columns)

    # Re-attach Source_URL / Dataset_Links after global merge.
    # merge_chunk_results keeps the first occurrence of each duplicate cluster,
    # so the index mapping is 1-to-1 for unique rows that survived.
    for i, row in enumerate(all_rows):
        if i < len(pre_merge_meta):
            meta = pre_merge_meta[i]
            if not row.get("Source_URL"):
                row["Source_URL"] = meta["Source_URL"]
            if not row.get("Dataset_Links") and meta["Dataset_Links"]:
                row["Dataset_Links"] = meta["Dataset_Links"]

    console.print(f"[dim]→ {len(all_rows)} unique rows after global dedup[/dim]")

    # ── Build DataFrame ───────────────────────────────────────────────────────
    df = pd.DataFrame(all_rows)
    for col in columns:
        if col not in df.columns:
            df[col] = "N/A"
    extra_cols = [c for c in df.columns if c not in columns]
    df = df[columns + extra_cols]

    # ── Save ────────────────────────────────────────────
    os.makedirs(os.path.dirname(output_path) or ".", exist_ok=True)
    try:
        df.to_csv(output_path, index=False, encoding="utf-8-sig")
    except PermissionError:
        from datetime import datetime
        stem_p, ext = os.path.splitext(output_path)
        output_path = f"{stem_p}_{datetime.now().strftime('%Y%m%d_%H%M%S')}{ext}"
        console.print(f"[yellow]⚠ Original file locked — saving as {os.path.basename(output_path)}[/yellow]")
        df.to_csv(output_path, index=False, encoding="utf-8-sig")
    console.print(f"\\n[green]✓ Saved {len(df)} rows → {output_path}[/green]")

    # ── Preview ───────────────────────────────────────────────────────────────
    try:
        tbl = Table(title=f"Results ({len(df)} rows)", show_lines=True, expand=False)
        preview_cols = list(df.columns[:8])
        for col in preview_cols:
            tbl.add_column(str(col), max_width=35, overflow="ellipsis")
        for _, row in df.head(8).iterrows():
            tbl.add_row(*[str(row[col])[:35] for col in preview_cols])
        console.print(tbl)
    except Exception as e:
        console.print(f"[yellow]⚠ Preview skipped: {e}[/yellow]")
        console.print(df.head(5).to_string())

    console.print(
        f"[bold green]Done![/bold green] {len(all_rows)} rows, {failed} source(s) failed.\n"
        f"Output: [cyan]{output_path}[/cyan]"
    )
    return df

# ── Run-level duplicate detection ─────────────────────────────────────────────
# Keyed by SHA-256 of raw PDF bytes.  Same paper reachable from two URLs will
# match here and be skipped the second time.
_SEEN_PDF_HASHES: set[str] = set()

def _register_pdf_hash(pdf_path: str) -> bool:
    """Return True if this PDF is a duplicate (already seen this run)."""
    try:
        with open(pdf_path, "rb") as fh:
            h = hashlib.sha256(fh.read()).hexdigest()
        if h in _SEEN_PDF_HASHES:
            console.print("  [dim]⊘ Duplicate PDF (same content) — skipping[/dim]")
            return True
        _SEEN_PDF_HASHES.add(h)
        return False
    except Exception:
        return False


# ── Dynamic paywall detection ──────────────────────────────────────────────
# Set at runtime by configure_paywall_detection() — called from pipeline.py.
# Empty defaults = no paywall blocking until LLM config is applied.

_HARD_PAYWALL_PATTERNS: list[str] = []

_PAYWALL_HTML_SIGNALS: list[str] = []


def configure_paywall_detection(paywall_domains: list[str], paywall_signals: list[str]) -> None:
    """
    Accept LLM-generated paywall config and apply it to this module.
    Called once per pipeline run from pipeline.py.
    """
    global _HARD_PAYWALL_PATTERNS, _PAYWALL_HTML_SIGNALS
    _HARD_PAYWALL_PATTERNS = [str(d).lower() for d in paywall_domains]
    _PAYWALL_HTML_SIGNALS = [str(s).lower() for s in paywall_signals]
    console.print(
        f"  [dim]Paywall config: {len(_HARD_PAYWALL_PATTERNS)} domain patterns, "
        f"{len(_PAYWALL_HTML_SIGNALS)} HTML signals[/dim]"
    )


def is_hard_paywalled(url: str) -> bool:
    """Return True if the URL is on a known hard-paywall publisher."""
    if not _HARD_PAYWALL_PATTERNS:
        return False   # no config = don't block anything
    u = url.lower()
    return any(p in u for p in _HARD_PAYWALL_PATTERNS)


def html_is_paywalled(html: str) -> bool:
    """Return True if the scraped HTML shows a paywall / subscription gate."""
    if not _PAYWALL_HTML_SIGNALS:
        return False   # no config = don't block anything
    snippet = html[:8000].lower()
    return sum(1 for sig in _PAYWALL_HTML_SIGNALS if sig in snippet) >= 2


def find_fulltext_fallback(url: str) -> list[str]:
    """
    Given a URL that yielded abstract-only or was blocked, return a prioritised
    list of free full-text alternatives to try next.

    Generic approach (no hardcoded sites):
      1. DOI-based redirect - doi.org resolves to publisher, try open access
      2. Search via DuckDuckGo to find alternative sources
      3. Title-based search for cached / mirrored versions
    """
    candidates: list[str] = []
    seen: set[str] = set()

    def _add(u: str) -> None:
        u = (u or "").strip()
        if u and u.startswith("http") and u not in seen:
            seen.add(u)
            candidates.append(u)

    # -- 1. DOI-based resolution (universal - works for any publisher) --------
    doi: str | None = None
    doi_m = re.search(r"(10\.\d{4,}/[^\s?#&\"'<>]+)", url)
    if doi_m:
        doi = doi_m.group(1).rstrip(".,)")

    if doi:
        console.print(f"  [dim]DOI redirect: {doi[:55]}[/dim]")
        try:
            r = requests.head(
                f"https://doi.org/{doi}",
                timeout=8, headers={"User-Agent": _random_ua()},
                allow_redirects=True,
            )
            if r.status_code == 200 and r.url != url:
                _add(r.url)
                base = r.url.rstrip("/")
                _add(base + "/pdf")
                _add(base + ".pdf")
        except Exception as exc:
            console.print(f"  [dim]  DOI redirect error: {exc}[/dim]")

    # -- 2. Search for alternative full-text via DuckDuckGo -------------------
    search_terms = []
    if doi:
        search_terms.append(f'"{doi}" full text pdf')
        search_terms.append(f'"{doi}" open access')
    else:
        url_parts = url.split("//")[-1].split("/")
        slug = " ".join(p.replace("-", " ").replace("_", " ")
                        for p in url_parts[1:4] if len(p) > 3 and not p.isdigit())
        if slug.strip():
            search_terms.append(f'{slug.strip()} full text')

    if search_terms:
        try:
            from ddgs import DDGS
            console.print(f"  [dim]Searching for alternative sources[/dim]")
            with DDGS() as ddgs:
                for query in search_terms[:2]:
                    hits = list(ddgs.text(query, max_results=3))
                    for hit in hits:
                        alt_url = hit.get("href", "")
                        if alt_url and alt_url != url:
                            _add(alt_url)
        except Exception as exc:
            console.print(f"  [dim]  Search fallback error: {exc}[/dim]")

    if candidates:
        console.print(f"  [cyan]{len(candidates)} alternative source(s) found[/cyan]")
    else:
        console.print("  [dim]  No alternative sources found for this URL[/dim]")
    return candidates

def simulate_human(page):
    """Simulate human-like mouse movements and scrolling."""
    try:
        vw = page.viewport_size
        if not vw:
            return

        # Random mouse movements (3-5 movements)
        for _ in range(random.randint(3, 5)):
            x = random.randint(100, vw["width"] - 100)
            y = random.randint(100, vw["height"] - 100)
            page.mouse.move(x, y)
            time.sleep(random.uniform(0.1, 0.3))

        # Random scroll
        scroll_y = random.randint(200, 800)
        page.mouse.wheel(0, scroll_y)
        time.sleep(random.uniform(0.5, 1.5))

        # Scroll back up a bit
        page.mouse.wheel(0, -random.randint(50, 200))
        time.sleep(random.uniform(0.3, 0.7))
    except Exception:
        pass  # don't crash on simulation failures

def download_pdf(url: str, page=None) -> str | None:
    """
    Download a PDF to a temp directory.
    If a Playwright `page` is provided, it uses page.request.get() to inherit 
    all stealth cookies/tokens. Otherwise falls back to requests.
    Returns path to downloaded file, or None on failure.
    """
    os.makedirs(PDF_TEMP_DIR, exist_ok=True)

    try:
        # Save to temp file
        base = url.split("?")[0].rstrip("/")
        filename = base.split("/")[-1]
        
        # If still empty or weird, use a hash or a generic name
        if not filename or filename.lower() in ("pdf", ""):
            filename = hashlib.md5(url.encode()).hexdigest()[:10]
            
        filename = re.sub(r'[^\w\-.]', '_', filename)
        if not filename.lower().endswith(".pdf"):
            filename += ".pdf"
        filepath = os.path.join(PDF_TEMP_DIR, filename)


        if page:
            # Use Playwright context to bypass bot protection
            resp = page.request.get(url, timeout=30000)
            if not resp.ok:
                console.print(f"  [yellow]⚠ PDF download failed: {resp.status} {resp.status_text} for url: {url}[/yellow]")
                return None
            
            ct = resp.headers.get("content-type", "").lower()
            body = resp.body()
            
            # Safety check: if we got HTML instead of PDF
            if "html" in ct or b"<html" in body.lower() or b"<!doctype" in body.lower():
                console.print(f"  [cyan]ℹ Detected download interstitial for {url}, waiting for trigger...[/cyan]")
                try:
                    with page.expect_download(timeout=20000) as download_info:
                        page.goto(url, wait_until="commit")
                    download = download_info.value
                    download.save_as(filepath)
                    console.print(f"  [green]✓ PDF downloaded via interstitial: {filename}[/green]")
                    return filepath
                except Exception as e:
                    console.print(f"  [yellow]⚠ Playwright download event failed: {e}[/yellow]")
                    return None

            if "pdf" not in ct and not url.lower().endswith(".pdf"):
                return None

            with open(filepath, "wb") as f:
                f.write(body)


        else:
            # Fallback to standard requests if called directly without a browser
            headers = {"User-Agent": _random_ua()}
            resp = requests.get(url, headers=headers, timeout=30, stream=True)
            resp.raise_for_status()

            # Safety check: if we got HTML instead of PDF, it's an interstitial/bot block
            ct = resp.headers.get("content-type", "").lower()
            if "html" in ct or "text" in ct:
                # Peek at content
                peek = next(resp.iter_content(512), b"")
                if b"<html" in peek.lower() or b"<!doctype" in peek.lower():
                    console.print(f"  [yellow]⚠ Received HTML instead of PDF for: {url}[/yellow]")
                    return None


            ct = resp.headers.get("Content-Type", "")
            if "pdf" not in ct.lower() and not url.lower().endswith(".pdf"):
                return None

            with open(filepath, "wb") as f:
                for chunk in resp.iter_content(8192):
                    f.write(chunk)

        console.print(f"  [dim]📥 Downloaded PDF: {filename}[/dim]")
        return filepath

    except Exception as e:
        console.print(f"  [yellow]⚠ PDF download failed: {e}[/yellow]")
        return None


def extract_pdf_text(pdf_path: str, truncate: bool = False) -> str:
    """Extract text from a PDF file using PyMuPDF.
    Returns '' if this PDF was already processed this run (content-hash duplicate).
    """
    # Duplicate gate — same bytes = same paper from a different URL
    if _register_pdf_hash(pdf_path):
        return ""
    try:
        doc = fitz.open(pdf_path)
        text_parts = []
        for page in doc:
            text_parts.append(page.get_text())
        doc.close()
        full_text = "\n".join(text_parts)
        if truncate and len(full_text) > MAX_TEXT_CHARS:
            return full_text[:MAX_TEXT_CHARS]
        return full_text
    except Exception as e:
        console.print(f"  [yellow]⚠ PDF text extraction failed: {e}[/yellow]")
        return ""


def extract_images_from_pdf(pdf_path: str, output_dir: str) -> int:
    """
    Extract images from PDF and save to output_dir.
    Returns number of images extracted.
    """
    os.makedirs(output_dir, exist_ok=True)
    count = 0
    try:
        doc = fitz.open(pdf_path)
        for i in range(len(doc)):
            for img_index, img in enumerate(doc.get_page_images(i)):
                xref = img[0]
                base_image = doc.extract_image(xref)
                image_bytes = base_image["image"]
                image_ext = base_image["ext"]
                
                filename = f"page_{i+1}_img_{img_index+1}.{image_ext}"
                filepath = os.path.join(output_dir, filename)
                
                with open(filepath, "wb") as f:
                    f.write(image_bytes)
                count += 1
        doc.close()
    except Exception as e:
        console.print(f"  [yellow]⚠ Image extraction failed: {e}[/yellow]")
    return count


# ── HTML scraping ────────────────────────────────────────────────────────────

# Minimum chars to consider as meaningful content
MIN_CONTENT_LENGTH = 300


def extract_html_text(html: str) -> str:
    """
    Extract article body text from HTML, stripping navigation and boilerplate.
    Handles academic sites (PubMed, arXiv, etc.) with special selectors.
    Falls back to full page text if article content is too short.
    """
    soup = BeautifulSoup(html, "lxml")

    # Remove unwanted elements
    for tag in soup.find_all(["nav", "header", "footer", "script", "style",
                               "aside", "iframe", "noscript", "form",
                               "button", "svg", "img", "label"]):
        tag.decompose()
        
    # Remove common academic noise and "open in new tab" junk
    # Note: string= replaces text= in BS4
    noise_patterns = re.compile(r"Open in a new tab|Find articles by|Correspondence:|Academic Editor|Roles:|Received |Revised |Accepted |Collection date|Share on|Follow us|Privacy Policy|Terms of Use", re.I)
    for noise in soup.find_all(string=noise_patterns):
        if hasattr(noise, 'parent') and noise.parent:
            noise.parent.decompose()



    # ── Step 1: Try meta tags first for title/abstract ────────────────────
    meta_parts = []
    
    # Title from <title> or og:title
    title_tag = soup.find("title")
    if title_tag:
        meta_parts.append(f"Title: {title_tag.get_text(strip=True)}")
    
    og_title = soup.find("meta", attrs={"property": "og:title"})
    if og_title and og_title.get("content"):
        meta_parts.append(f"Title: {og_title['content']}")

    # Description from meta description or og:description
    for meta in soup.find_all("meta"):
        name = meta.get("name", "").lower()
        prop = meta.get("property", "").lower()
        content = meta.get("content", "")
        if content and name in ("description", "dc.description", "citation_abstract"):
            meta_parts.append(f"Abstract: {content}")
        elif content and prop in ("og:description",):
            meta_parts.append(f"Description: {content}")
        elif content and name == "citation_title":
            meta_parts.append(f"Title: {content}")
        elif content and name == "citation_author":
            meta_parts.append(f"Author: {content}")
        elif content and name == "citation_doi":
            meta_parts.append(f"DOI: {content}")
        elif content and name == "citation_date":
            meta_parts.append(f"Date: {content}")
        elif content and name in ("keywords", "citation_keywords"):
            meta_parts.append(f"Keywords: {content}")

    meta_text = "\n".join(meta_parts)

    # ── Step 2: Try academic site-specific selectors ──────────────────────
    article_text = ""

    # PubMed / PMC / Generic Modern Academic Site
    article_tag = soup.find("article")
    pmc_body = soup.find("div", id="body") or soup.find("div", class_="body") or soup.find("div", class_="jig-ncbi-article")
    
    if article_tag:
        article_text = article_tag.get_text(separator=" ", strip=True)
    elif pmc_body:
        article_text = pmc_body.get_text(separator=" ", strip=True)
    
    # If using separator=" ", we might need to fix joined words, 
    # but it's often better than thousands of newlines.
    # Let's try to restore some paragraph structure while removing tag-spam.
    if article_text:
        # Replace multiple spaces with one
        article_text = re.sub(r'[ \t]+', ' ', article_text)
        # Restore some newlines for sections
        article_text = re.sub(r'([.?!])\s+([A-Z])', r'\1\n\n\2', article_text)


    # arXiv
    if not article_text or len(article_text) < MIN_CONTENT_LENGTH:
        arxiv_abs = soup.find("blockquote", class_="abstract")
        if arxiv_abs:
            article_text = arxiv_abs.get_text(separator="\n", strip=True)

    # Generic: look for abstract sections
    if not article_text or len(article_text) < MIN_CONTENT_LENGTH:
        abstract_section = (
            soup.find("section", id=re.compile(r"abstract", re.I))
            or soup.find("div", id=re.compile(r"abstract", re.I))
            or soup.find("div", class_=re.compile(r"abstract", re.I))
            or soup.find("p", class_=re.compile(r"abstract", re.I))
        )
        if abstract_section:
            article_text = abstract_section.get_text(separator="\n", strip=True)

    if not article_text or len(article_text) < MIN_CONTENT_LENGTH:
        article = (
            soup.find("article")
            or soup.find("main")
            or soup.find("div", class_=re.compile(r"(article|content|paper|fulltext|full-text|post|entry|main)", re.I))
            or soup.find("div", id=re.compile(r"(article|content|paper|fulltext|full-text|post|entry|main)", re.I))
            or soup.find("div", role="main")
            or soup.find("section", class_=re.compile(r"(content|main)", re.I))
        )

        if article:
            candidate = article.get_text(separator="\n", strip=True)
            # Only use it if substantial
            if len(candidate) > len(article_text):
                article_text = candidate

    # ── Step 4: Fallback to full body text ────────────────────────────────
    if not article_text or len(article_text) < MIN_CONTENT_LENGTH:
        body = soup.find("body")
        if body:
            article_text = body.get_text(separator="\n", strip=True)
        else:
            article_text = soup.get_text(separator="\n", strip=True)

    # ── Step 5: Combine meta + article text ───────────────────────────────
    if meta_text:
        full_text = meta_text + "\n\n---\n\n" + article_text
    else:
        full_text = article_text

    # Clean up excessive whitespace
    full_text = re.sub(r"\n{3,}", "\n\n", full_text)
    full_text = re.sub(r" {2,}", " ", full_text)

    # Return the full unstructured text to be chunked safely by the LLM pipeline
    return full_text


def extract_html_artifacts(html: str, base_url: str) -> dict:
    """
    Extract images and links from HTML content.
    Returns a dict with 'images' (list of urls) and 'all_links' (list of {text, url}).
    """
    soup = BeautifulSoup(html, "lxml")
    artifacts = {"images": [], "all_links": []}
    
    # Extract images
    for img in soup.find_all("img"):
        src = img.get("src") or img.get("data-src")
        if src:
            if src.startswith("//"):
                src = "https:" + src
            elif src.startswith("/"):
                src = urljoin(base_url, src)
            if src.startswith("http"):
                artifacts["images"].append(src)
                
    # Extract all links
    for a in soup.find_all("a", href=True):
        href = a["href"]
        text = a.get_text(strip=True) or "[No Text]"
        if href.startswith("/"):
            from urllib.parse import urljoin
            href = urljoin(base_url, href)
        if href.startswith("http"):
            artifacts["all_links"].append({"text": text, "url": href})
            
    return artifacts

def _find_pdf_links_in_html(html: str, base_url: str) -> list[str]:
    """
    Find PDF download links from an article HTML page.
    Uses only generic strategies — no hardcoded site-specific URL rewriting.
    """

    soup = BeautifulSoup(html, "lxml")
    pdf_urls = []
    seen = set()

    # Strategy 1 (highest priority): meta citation_pdf_url tag
    # This is a standard academic HTML meta tag used by many publishers
    meta_pdf = soup.find("meta", attrs={"name": "citation_pdf_url"})
    if meta_pdf and meta_pdf.get("content"):
        pdf_urls.append(meta_pdf["content"])
        seen.add(meta_pdf["content"])

    # Strategy 2: URL path heuristics (generic, not site-specific)
    # Many sites use /pdf/ or .pdf suffixes on the same path
    base_clean = base_url.rstrip("/")
    for suffix in ["/pdf", "/pdf/", ".pdf"]:
        candidate = base_clean + suffix
        if candidate not in seen:
            # Only add if it looks different from the base URL
            if candidate.lower() != base_url.lower():
                seen.add(candidate)
                pdf_urls.append(candidate)

    # Strategy 3: URL path rewriting (/abs/ → /pdf/, /view/ → /download/)
    url_lower = base_url.lower()
    rewrite_rules = [
        ("/abs/", "/pdf/"),
        ("/view/", "/download/"),
        ("/article/", "/article/pdf/"),
        ("/full/", "/pdf/"),
        ("/html/", "/pdf/"),
    ]
    for old, new in rewrite_rules:
        if old in url_lower:
            rewritten = url_lower.replace(old, new, 1)
            if not rewritten.endswith(".pdf"):
                rewritten += ".pdf"
            if rewritten not in seen:
                seen.add(rewritten)
                pdf_urls.append(rewritten)

    # Strategy 4: scan all links for PDF hrefs and download-like text
    pdf_link_keywords = [
        "pdf", "download pdf", "full text pdf", "view pdf",
        "download article", "download full", "get pdf",
        "full text", "télécharger", "descargar",  # French/Spanish
    ]
    for a in soup.find_all("a", href=True):
        href = a["href"]
        text = a.get_text(strip=True).lower()

        # Direct PDF link
        if href.lower().endswith(".pdf"):
            full_url = urljoin(base_url, href)
            if full_url not in seen:
                seen.add(full_url)
                pdf_urls.append(full_url)

        # Links with download/PDF text
        if any(kw in text for kw in pdf_link_keywords):
            full_url = urljoin(base_url, href)
            if full_url not in seen:
                seen.add(full_url)
                pdf_urls.append(full_url)

        # Links with PDF in the URL itself (e.g. /getpdf, /renderpdf)
        if "pdf" in href.lower() and href.lower() != base_url.lower():
            full_url = urljoin(base_url, href)
            if full_url not in seen:
                seen.add(full_url)
                pdf_urls.append(full_url)

    return pdf_urls


def scrape_url(url: str, skip_pdf: bool = False) -> dict | None:
    """
    PDF-first scraping dispatcher.

    Strategy:
    1. If URL is already a PDF → download & extract (unless skip_pdf is True)
    2. Fetch HTML page → look for PDF links → download & extract the PDF (unless skip_pdf is True)
    3. Fall back to HTML text if no PDF found

    Returns:
        dict with keys: {text, html, pdf_path, url, method}
        or None on complete failure.
    """
    console.print(f"  [cyan]📄 Scraping: {url[:80]}...[/cyan]")

    # ── Early exit: known hard-paywall domain ────────────────────────────
    if is_hard_paywalled(url):
        console.print(f"  [yellow]⊘ Skipped (hard paywall publisher): {url[:70]}[/yellow]")
        return None

    # Random delay for politeness
    time.sleep(random.uniform(*REQUEST_DELAY))

    result = {
        "url": url,
        "text": "",
        "html": "",
        "pdf_path": None,
        "method": "none",
    }

    # ── Path A: URL is already a PDF ─────────────────────────────────────

    is_pdf = url.lower().endswith(".pdf") or "/pdf/" in url.lower()

    if is_pdf and not skip_pdf:

        # Try simple requests first
        pdf_path = download_pdf(url)
        
        # Fallback to Playwright if simple request fails
        if not pdf_path:
            console.print("  [dim]Retrying PDF download with Playwright stealth...[/dim]")
            browser = init_browser()
            context = browser.new_context(
                user_agent=_random_ua(),
                viewport={"width": random.randint(1200, 1920), "height": random.randint(800, 1080)},
                locale="en-US",
            )
            page = context.new_page()
            page.add_init_script(STEALTH_JS)
            pdf_path = download_pdf(url, page=page)
            context.close()

        if pdf_path:
            text = extract_pdf_text(pdf_path)  # returns "" if duplicate hash
            if text and len(text) > 200:
                result["text"] = text
                result["pdf_path"] = pdf_path
                result["method"] = "pdf"
                console.print(f"  [green]✓ PDF extracted ({len(text)} chars)[/green]")
                return result
            elif not text and pdf_path:
                console.print("  [dim]⊘ PDF is duplicate — trying next source[/dim]")
                return None

    # ── Path B: Fetch HTML, then look for PDF links ──────────────────────
    # Use Playwright for initial HTML scrape to get a page context
    browser = init_browser()
    context = browser.new_context(
        user_agent=_random_ua(),
        viewport={"width": random.randint(1200, 1920), "height": random.randint(800, 1080)},
        locale="en-US",
    )
    page = context.new_page()
    html = ""

    try:
        try:
            page.add_init_script(STEALTH_JS)
            page.goto(url, wait_until="domcontentloaded", timeout=30000)
            simulate_human(page) # Simulate human behavior after navigation
            time.sleep(random.uniform(1, 3)) # Wait a bit for dynamic content
            html = page.content()
        except Exception as e:
            console.print(f"  [yellow]⚠ Playwright initial HTML scrape failed: {e}[/yellow]")
            # Fallback to requests if Playwright fails for initial HTML
            html = _scrape_with_requests(url)

        if html:
            result["html"] = html

            # ── HTML paywall gate: bail out fast if page is gated ──────
            if html_is_paywalled(html):
                console.print(f"  [yellow]⊘ Paywall detected in HTML body — skipping[/yellow]")
                return None

            # Try to find PDF links in the HTML page
            pdf_links = []
            if not skip_pdf:
                pdf_links = _find_pdf_links_in_html(page.content() if page else html, page.url if page else url)

            if pdf_links:
                console.print(f"  [cyan]🔗 Found {len(pdf_links)} PDF link(s), downloading...[/cyan]")
                for pdf_url in pdf_links[:3]:  # try up to 3 links
                    console.print(f"    [dim]Trying: {pdf_url[:80]}...[/dim]")

                    # Pass `page` so the download uses Playwright's cookies/context
                    pdf_path = download_pdf(pdf_url, page=page)

                    if pdf_path:
                        text = extract_pdf_text(pdf_path)
                        if text and len(text) > 500:
                            result["text"] = text
                            result["pdf_path"] = pdf_path
                            result["method"] = "pdf"
                            console.print(f"  [green]✓ PDF extracted ({len(text)} chars)[/green]")
                            return result  # context.close() runs in finally below

            # No PDF found — fall back to HTML text
            text = extract_html_text(html)
            if text and len(text) > MIN_CONTENT_LENGTH:
                result["text"] = text
                result["method"] = "html"

                # Extract images and all links
                artifacts = extract_html_artifacts(html, url)
                result["images"] = artifacts["images"]
                result["all_links"] = artifacts["all_links"]

                console.print(f"  [green]✓ Content extracted via html ({len(text)} chars)[/green]")

            else:
                console.print(f"  [yellow]⚠ HTML extraction yielded too little text ({len(text)} chars)[/yellow]")
                return None

    finally:
        # Fix #5: Always close the context regardless of how we exit the block.
        # Previously, early returns on PDF success left the context open.
        context.close()

    if result["text"]:
        return result

    # ── Path C: Check Content-Type header as last resort ─────────────────
    try:
        headers = {"User-Agent": _random_ua()}
        head_resp = requests.head(url, headers=headers, timeout=10, allow_redirects=True)
        ct = head_resp.headers.get("Content-Type", "")
        if "pdf" in ct.lower():
            pdf_path = download_pdf(url)
            if pdf_path:
                text = extract_pdf_text(pdf_path)
                if text:
                    result["text"] = text
                    result["pdf_path"] = pdf_path
                    result["method"] = "pdf"
                    console.print(f"  [green]✓ PDF extracted via Content-Type fallback ({len(text)} chars)[/green]")
                    return result
    except Exception:
        pass

    console.print(f"  [red]✗ Failed to extract content from {url[:60]}[/red]")
    return None

def scrape_local_pdf(path: str) -> dict | None:
    """
    Process a local PDF file (not from the web).
    Returns same dict format as scrape_url().
    """
    path = str(path)
    if not os.path.isfile(path):
        console.print(f"  [red]✗ File not found: {path}[/red]")
        return None
    console.print(f"  [cyan]📄 Reading local PDF: {os.path.basename(path)}[/cyan]")
    text = extract_pdf_text(path)
    if not text or len(text.strip()) < 200:
        from extractor import extract_pdf_text_ocr
        console.print("  [yellow]Text too short, trying OCR…[/yellow]")
        text = extract_pdf_text_ocr(path)
    if not text or len(text.strip()) < 100:
        console.print(f"  [red]✗ Could not extract text from {path}[/red]")
        return None

    # Extract images into data/images/<stem>/
    stem = os.path.splitext(os.path.basename(path))[0]
    images_dir = os.path.join("data", "images", stem)
    n_imgs = extract_images_from_pdf(path, images_dir)
    if n_imgs:
        console.print(f"  [dim]📷 Extracted {n_imgs} image(s) → {images_dir}[/dim]")

    console.print(f"  [green]✓ Extracted {len(text)} chars[/green]")
    return {"url": path, "text": text, "html": "", "pdf_path": path,
            "method": "local_pdf", "images_dir": images_dir if n_imgs else None}
"""
searcher.py — Web search with dynamic topic-aware domain configuration.
Searches DuckDuckGo and filters/prioritises results based on LLM-generated
domain config set at pipeline start via configure_for_topic().
"""

import random
from ddgs import DDGS
from rich.console import Console

import config as cfg

REQUEST_DELAY = cfg.REQUEST_DELAY
console = Console()

# ── Dynamic domain configuration ─────────────────────────────────────────────
# Set at runtime by configure_for_topic() — called from pipeline.py.
# Default to empty so the searcher works even without LLM config (accepts all).

_relevant_domains: list[str] = []
_blocked_domains: list[str] = []
_priority_tiers: list[list[str]] = []  # tier 0 = best, tier N = lowest


def configure_for_topic(domain_config: dict) -> None:
    """
    Accept LLM-generated domain config and apply it to this module.
    Called once per pipeline run from pipeline.py.
    """
    global _relevant_domains, _blocked_domains, _priority_tiers
    _relevant_domains = [str(d).lower() for d in domain_config.get("relevant_domains", [])]
    _blocked_domains = [str(d).lower() for d in domain_config.get("blocked_domains", [])]
    raw_tiers = domain_config.get("priority_tiers", [])
    _priority_tiers = [
        [str(d).lower() for d in tier] for tier in raw_tiers if isinstance(tier, list)
    ]
    console.print(
        f"  [dim]Domain config: {len(_relevant_domains)} relevant, "
        f"{len(_blocked_domains)} blocked, {len(_priority_tiers)} priority tiers[/dim]"
    )


# ── URL path patterns that indicate non-content pages (universal) ─────────────

JUNK_URL_PATTERNS = [
    "/tag/", "/category/", "/blog/", "/news/", "/about/",
    "/contact/", "/login", "/signup", "/cart", "/product/", "/shop/",
    "/search?", "/topic/", "/author/", "/press-release/",
]


# ── Query builder — generic fallback ─────────────────────────────────────────

def build_search_queries(prompt: str) -> list[str]:
    """
    Generic fallback queries when no LLM-generated queries are available.
    No hardcoded site: operators — works for any topic.
    """
    return [
        f'{prompt}',
        f'{prompt} research study data',
        f'{prompt} review article results',
        f'{prompt} dataset doi abstract',
    ]


# ── URL filters ───────────────────────────────────────────────────────────────

def _is_blocked(url: str) -> bool:
    """Return True if this URL should never be included."""
    u = url.lower()
    if any(domain in u for domain in _blocked_domains):
        return True
    if any(pattern in u for pattern in JUNK_URL_PATTERNS):
        return True
    return False


def _is_relevant_domain(url: str) -> bool:
    """Return True if the URL belongs to a known relevant domain for this topic."""
    u = url.lower()
    return any(domain in u for domain in _relevant_domains)


def _priority(url: str) -> int:
    """
    Lower score = higher priority.
    Uses LLM-generated priority tiers. Falls back to relevant/unknown scoring.
    """
    u = url.lower()

    # Check LLM-generated tiers first
    for tier_idx, tier_domains in enumerate(_priority_tiers):
        if any(d in u for d in tier_domains):
            return tier_idx

    # Known relevant domain but not in any tier → moderate priority
    if _is_relevant_domain(url):
        return len(_priority_tiers) + 1  # after all defined tiers

    return 99   # unknown domain — lowest priority


# ── Main search function ──────────────────────────────────────────────────────

def search_sources(query: str, limit: int = 10,
                   custom_queries: list[str] | None = None) -> list[dict]:
    """
    Search DuckDuckGo for sources on the topic.

    custom_queries – LLM-generated queries tailored to the topic.
                     If omitted, falls back to build_search_queries().

    Domain filtering:
      - Always blocks _blocked_domains (set by LLM via configure_for_topic).
      - Any non-blocked URL is accepted — works for any topic.
      - is_relevant / priority flags are set for informational sorting.
    """
    results: list[dict] = []
    seen_urls: set[str] = set()
    queries = custom_queries if custom_queries else build_search_queries(query)

    # Collect a small buffer (2x limit) so sorting has material to work with
    per_query = max(limit // len(queries) + 2, 2)
    buffer_cap = limit * 2

    for q in queries:
        if len(results) >= buffer_cap:
            break
        try:
            console.print(f"  [dim]🔍 {q[:90]}[/dim]")
            with DDGS() as ddgs:
                hits = list(ddgs.text(q, max_results=per_query))

            for hit in hits:
                url = hit.get("href", "")
                if not url or url in seen_urls:
                    continue
                if _is_blocked(url):
                    console.print(f"  [dim]✗ Skipped (blocked): {url[:70]}[/dim]")
                    continue

                seen_urls.add(url)
                results.append({
                    "title":        hit.get("title", ""),
                    "url":          url,
                    "snippet":      hit.get("body", ""),
                    "is_relevant":  _is_relevant_domain(url),
                    "priority":     _priority(url),
                })

        except Exception as e:
            console.print(f"  [yellow]⚠ Search query failed: {e}[/yellow]")

    # Sort best sources first, then enforce hard limit
    results.sort(key=lambda r: r["priority"])
    results = results[:limit]

    if not results:
        console.print("[yellow]⚠ No sources found. Try a more specific topic.[/yellow]")
    else:
        relevant_count = sum(1 for r in results if r["is_relevant"])
        console.print(
            f"  [green]✓ {len(results)} source(s) — "
            f"{relevant_count} from known relevant domains[/green]"
        )

    return results
